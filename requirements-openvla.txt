# Isolated environment for the OpenVLA transfer pipeline.
#
# OpenVLA's trust_remote_code modeling/processing files (modeling_prismatic.py,
# processing_prismatic.py) target transformers ~4.40. They are NOT compatible
# with transformers 5.x (PrismaticImageProcessor lacks `valid_kwargs`, etc.), so
# the OpenVLA pipeline MUST run in its own venv, separate from the project's main
# requirements.txt env (which uses transformers 5.x for the Qwen+FlowMatching path).
#
# Create with:
#   python -m venv .venv-openvla && . .venv-openvla/Scripts/activate   # Windows: .venv-openvla\Scripts\activate
#   pip install -r requirements-openvla.txt
#
# Train on the lab GPU (>=16GB for 4-bit, >=24GB for bf16):
#   python scripts/train_openvla.py --config configs/openvla.yaml --init pretrained
#   python scripts/train_openvla.py --config configs/openvla.yaml --init scratch     # control arm

torch>=2.2.0,<2.6
torchvision
transformers==4.40.1          # pinned: OpenVLA remote code requires this API
tokenizers>=0.19,<0.20
timm==0.9.10                  # DINOv2 + SigLIP vision backbones
# accelerate/peft are PINNED, not floored: they must stay in the transformers 4.40.1
# era. A floor (>=) silently keeps whatever a host already ships (Colab: accelerate
# 1.14), and modern accelerate skips the dispatch-hook path for 4-bit bnb models —
# it moves them with .to() instead, which 4.40.1's .to() rejects outright, so
# from_pretrained dies AFTER the full 15GB load. Bump these only with transformers.
accelerate==0.29.3
peft==0.11.1                  # LoRA on the OpenVLA backbone
bitsandbytes>=0.43.0          # 4-bit QLoRA to fit smaller GPUs; kept current on
                              # purpose — it is a compiled CUDA lib and needs to
                              # match the host's driver, unlike the two above
numpy>=1.26.0
Pillow>=10.0.0
PyYAML>=6.0
# flash-attn>=2.5.0           # optional, Linux/CUDA only — big speedup, skip on Windows
